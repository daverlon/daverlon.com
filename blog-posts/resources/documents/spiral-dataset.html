y = np.zeros(N*K, dtype='uint8') # class labels
for j in range(K):
  ix = range(N*j,N*(j+1))
  r = np.linspace(0.0,1,N) # radius
  t = np.linspace(j*4,(j+1)*4,N) + np.random.randn(N)*0.2 # theta
  X[ix] = np.c_[r*np.sin(t), r*np.cos(t)]
  y[ix] = j

plt.show()

def accuracy_fn(pred_logits, target_class):
    y = (torch.argmax(torch.softmax(pred_logits, dim=1), dim=1)==target_class).detach().numpy()
    y = np.sum(y) / len(target_class)
    return y*100

class spiralNet(nn.Module):
    def __init__(self):
        super().__init__()
        self.L1 = nn.Linear(in_features=2, out_features=12)
        self.L2 = nn.Linear(in_features=12, out_features=20)
        self.L3 = nn.Linear(in_features=20, out_features=20)
        self.L4 = nn.Linear(in_features=20, out_features=12)
        self.relu = nn.ReLU()
        
    def forward(self, x):
        x = self.L1(x)
        x = self.relu(x)
        x = self.L2(x)
        x = self.relu(x)
        x = self.L3(x)
        x = self.relu(x)
        x = self.L4(x)
        return x
    
model1 = spiralNet()

    acc = accuracy_fn(logits, y_train)
    if epoch % 10 == 0: t.set_description(f'accuracy: {acc}')

    loss = loss_fn(logits, y_train)
    
    loss.backward()
    
    optim.step()
